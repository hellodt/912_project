　　int gen(enum fct x, /* f */
　　　　　　int y, /* l */
　　　　　　int z /* a */
　　　　　　)
　　{
　　　if(cx>cxmax)
　　　{
　　　　printf("Program too long"); /* 程序过长 */
　　　　return -1;
　　　}
　　　code[cx].f=x;
　　　code[cx].l=y;
　　　code[cx].a=z;
　　　cx++;
　　　return 0;
　　}
