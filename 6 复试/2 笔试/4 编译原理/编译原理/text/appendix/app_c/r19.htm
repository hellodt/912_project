INTERVAL hilo(a, b, c, d)
　　　　double a, b, c, d; {
　　　　/* return the smallest interval */
　　　　/* containing a, b, c, and d used */
　　　　/* by *, / routines */
　　　　INTERVAL v;

　　　if (a>b) { v.hi = a; v.lo = b; }
　　　　else { v.hi = b; v.lo = a; }

　　　if (c>d) {
　　　　　　　　if (c>v.hi) v.hi= c;
　　　　　　　　if (d<v.lo) v.lo= d;
　　　　　　　　}
　　　else {
　　　　　　if (d>v.hi) v.hi= d;
　　　　　　if (c<v.lo) v.lo= c;
　　　　　　}
　　　return(v);
　　}

INTERVAL vmul(a, b, v)
double a, b;
INTERVAL v; {
　　return(hilo(a*v.hi, a*v.lo, b*v.hi, b*v.lo));
}

dckeck(v)
INTERVAL v; {
　　if (v.hi>=0. && v.lo<=0. ){
　　　printf("divisor interval contains 0.\n");
　　　return(1);
　　}
return(0);
}

INTERVAL vdiv(a, b, v)
double a, b;
INTERVAL v; {
　　return( hilo(a/v.hi, a/v.lo, b/v.hi, b/v.lo));
}
